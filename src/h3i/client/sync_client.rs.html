53/// Connect to a server and execute provided actions.
54///
55/// Constructs a socket and [quiche::Connection] based on the provided `args`,
56/// then iterates over `actions`.
57///
58/// Returns a [ConnectionSummary] on success, [ClientError] on failure.
59pub fn connect(
60    args: &Config, actions: &[Action],
61) -> std::result::Result<ConnectionSummary, ClientError> {
62    let mut buf = [0; 65535];
63    let mut out = [0; MAX_DATAGRAM_SIZE];
64
65    // Setup the event loop.
66    let mut poll = mio::Poll::new().unwrap();
67    let mut events = mio::Events::with_capacity(1024);
68
69    // We'll only connect to one server.
70    let connect_url = if !args.omit_sni {
71        args.host_port.split(':').next()
72    } else {
73        None
74    };
75
76    // Resolve server address.
77    let peer_addr = if let Some(addr) = &args.connect_to {
78        addr.parse().expect("--connect-to is expected to be a string containing an IPv4 or IPv6 address with a port. E.g. 192.0.2.0:443")
79    } else {
80        let x = format!("https://{}", args.host_port);
81        url::Url::parse(&x)
82            .unwrap()
83            .to_socket_addrs()
84            .unwrap()
85            .next()
86            .unwrap()
87    };
88
89    // Bind to INADDR_ANY or IN6ADDR_ANY depending on the IP family of the
90    // server address. This is needed on macOS and BSD variants that don't
91    // support binding to IN6ADDR_ANY for both v4 and v6.
92    let bind_addr = match peer_addr {
93        std::net::SocketAddr::V4(_) => format!("0.0.0.0:{}", args.source_port),
94        std::net::SocketAddr::V6(_) => format!("[::]:{}", args.source_port),
95    };
96
97    // Create the UDP socket backing the QUIC connection, and register it with
98    // the event loop.
99    let mut socket =
100        mio::net::UdpSocket::bind(bind_addr.parse().unwrap()).unwrap();
101    poll.registry()
102        .register(&mut socket, mio::Token(0), mio::Interest::READABLE)
103        .unwrap();
104
105    // Create the configuration for the QUIC connection.
106    let mut config = quiche::Config::new(1).unwrap();
107
108    config.verify_peer(args.verify_peer);
109    config.set_application_protos(&[b"h3"]).unwrap();
110    config.set_max_idle_timeout(args.idle_timeout);
111    config.set_max_recv_udp_payload_size(MAX_DATAGRAM_SIZE);
112    config.set_max_send_udp_payload_size(MAX_DATAGRAM_SIZE);
113    config.set_initial_max_data(10_000_000);
114    config
115        .set_initial_max_stream_data_bidi_local(args.max_stream_data_bidi_local);
116    config.set_initial_max_stream_data_bidi_remote(
117        args.max_stream_data_bidi_remote,
118    );
119    config.set_initial_max_stream_data_uni(args.max_stream_data_uni);
120    config.set_initial_max_streams_bidi(args.max_streams_bidi);
121    config.set_initial_max_streams_uni(args.max_streams_uni);
122    config.set_disable_active_migration(true);
123    config.set_active_connection_id_limit(0);
124
125    config.set_max_connection_window(args.max_window);
126    config.set_max_stream_window(args.max_stream_window);
127
128    let mut keylog = None;
129
130    if let Some(keylog_path) = std::env::var_os("SSLKEYLOGFILE") {
131        let file = std::fs::OpenOptions::new()
132            .create(true)
133            .append(true)
134            .open(keylog_path)
135            .unwrap();
136
137        keylog = Some(file);
138
139        config.log_keys();
140    }
141
142    config.grease(false);
143
144    let mut app_proto_selected = false;
145
146    // Generate a random source connection ID for the connection.
147    let mut scid = [0; quiche::MAX_CONN_ID_LEN];
148    rand::RngCore::fill_bytes(&mut rand::thread_rng(), &mut scid);
149
150    let scid = quiche::ConnectionId::from_ref(&scid);
151
152    let local_addr = socket.local_addr().unwrap();
153
154    // Create a QUIC connection and initiate handshake.
155    let mut conn =
156        quiche::connect(connect_url, &scid, local_addr, peer_addr, &mut config)
157            .unwrap();
158
159    if let Some(keylog) = &mut keylog {
160        if let Ok(keylog) = keylog.try_clone() {
161            conn.set_keylog(Box::new(keylog));
162        }
163    }
164
165    if let Some(dir) = std::env::var_os("QLOGDIR") {
166        let id = format!("{scid:?}");
167        let writer = make_qlog_writer(&dir, "client", &id);
168
169        conn.set_qlog(
170            std::boxed::Box::new(writer),
171            "h3i-client qlog".to_string(),
172            format!("{} id={}", "quiche-client qlog", id),
173        );
174    }
175
176    log::info!(
177        "connecting to {:} from {:} with scid {:?}",
178        peer_addr,
179        socket.local_addr().unwrap(),
180        scid,
181    );
182
183    let (write, send_info) = conn.send(&mut out).expect("initial send failed");
184
185    while let Err(e) = socket.send_to(&out[..write], send_info.to) {
186        if e.kind() == std::io::ErrorKind::WouldBlock {
187            log::debug!(
188                "{} -> {}: send() would block",
189                socket.local_addr().unwrap(),
190                send_info.to
191            );
192            continue;
193        }
194
195        return Err(ClientError::Other(format!("send() failed: {e:?}")));
196    }
197
198    let app_data_start = std::time::Instant::now();
199
200    let mut action_iter = actions.iter();
201    let mut wait_duration = None;
202    let mut wait_instant = None;
203
204    let mut client_variant = ClientVariant::Sync {
205        stream_map: StreamMap::default(),
206        stream_parser_map: StreamParserMap::default(),
207    };
208
209    let mut waiting_for = WaitingFor::default();
210
211    loop {
212        let actual_sleep = match (wait_duration, conn.timeout()) {
213            (Some(wait), Some(timeout)) => {
214                #[allow(clippy::comparison_chain)]
215                if timeout < wait {
216                    // shave some off the wait time so it doesn't go longer
217                    // than user really wanted.
218                    let new = wait - timeout;
219                    wait_duration = Some(new);
220                    Some(timeout)
221                } else if wait < timeout {
222                    Some(wait)
223                } else {
224                    // same, so picking either doesn't matter
225                    Some(timeout)
226                }
227            },
228            (None, Some(timeout)) => Some(timeout),
229            (Some(wait), None) => Some(wait),
230            _ => None,
231        };
232
233        log::debug!("actual sleep is {:?}", actual_sleep);
234        poll.poll(&mut events, actual_sleep).unwrap();
235
236        // If the event loop reported no events, run a belt and braces check on
237        // the quiche connection's timeouts.
238        if events.is_empty() {
239            log::debug!("timed out");
240
241            conn.on_timeout();
242        }
243
244        // Read incoming UDP packets from the socket and feed them to quiche,
245        // until there are no more packets to read.
246        for event in &events {
247            let socket = match event.token() {
248                mio::Token(0) => &socket,
249
250                _ => unreachable!(),
251            };
252
253            let local_addr = socket.local_addr().unwrap();
254            'read: loop {
255                let (len, from) = match socket.recv_from(&mut buf) {
256                    Ok(v) => v,
257
258                    Err(e) => {
259                        // There are no more UDP packets to read on this socket.
260                        // Process subsequent events.
261                        if e.kind() == std::io::ErrorKind::WouldBlock {
262                            break 'read;
263                        }
264
265                        return Err(ClientError::Other(format!(
266                            "{local_addr}: recv() failed: {e:?}"
267                        )));
268                    },
269                };
270
271                let recv_info = quiche::RecvInfo {
272                    to: local_addr,
273                    from,
274                };
275
276                // Process potentially coalesced packets.
277                let _read = match conn.recv(&mut buf[..len], recv_info) {
278                    Ok(v) => v,
279
280                    Err(e) => {
281                        log::debug!("{}: recv failed: {:?}", local_addr, e);
282                        continue 'read;
283                    },
284                };
285            }
286        }
287
288        log::debug!("done reading");
289
290        if conn.is_closed() {
291            log::info!(
292                "connection closed with error={:?} did_idle_timeout={}, stats={:?} path_stats={:?}",
293                conn.peer_error(),
294                conn.is_timed_out(),
295                conn.stats(),
296                conn.path_stats().collect::<Vec<quiche::PathStats>>(),
297            );
298
299            if !conn.is_established() {
300                log::info!(
301                    "connection timed out after {:?}",
302                    app_data_start.elapsed(),
303                );
304
305                return Err(ClientError::HandshakeFail);
306            }
307
308            break;
309        }
310
311        // Create a new application protocol session once the QUIC connection is
312        // established.
313        if (conn.is_established() || conn.is_in_early_data()) &&
314            !app_proto_selected
315        {
316            app_proto_selected = true;
317        }
318
319        if app_proto_selected {
320            #[allow(irrefutable_let_patterns)]
321            let ClientVariant::Sync {
322                ref mut stream_parser_map,
323                ..
324            } = client_variant
325            else {
326                panic!("async ClientVariant in sync code")
327            };
328
329            check_duration_and_do_actions(
330                &mut wait_duration,
331                &mut wait_instant,
332                &mut action_iter,
333                &mut conn,
334                &mut waiting_for,
335                stream_parser_map,
336            );
337
338            let mut wait_cleared = false;
339            for response in parse_streams(&mut conn, &mut client_variant) {
340                let stream_id = response.stream_id;
341
342                if let StreamEventType::Finished = response.event_type {
343                    waiting_for.clear_waits_on_stream(stream_id);
344                } else {
345                    waiting_for.remove_wait(response);
346                }
347
348                wait_cleared = true;
349            }
350
351            if wait_cleared {
352                #[allow(irrefutable_let_patterns)]
353                let ClientVariant::Sync {
354                    ref mut stream_parser_map,
355                    ..
356                } = client_variant
357                else {
358                    unimplemented!("async ClientVariant in sync code")
359                };
360
361                check_duration_and_do_actions(
362                    &mut wait_duration,
363                    &mut wait_instant,
364                    &mut action_iter,
365                    &mut conn,
366                    &mut waiting_for,
367                    stream_parser_map,
368                );
369            }
370        }
371
372        // Provides as many CIDs as possible.
373        while conn.scids_left() > 0 {
374            let (scid, reset_token) = generate_cid_and_reset_token();
375
376            if conn.new_scid(&scid, reset_token, false).is_err() {
377                break;
378            }
379        }
380
381        // Generate outgoing QUIC packets and send them on the UDP socket, until
382        // quiche reports that there are no more packets to be sent.
383        let sockets = vec![&socket];
384
385        for socket in sockets {
386            let local_addr = socket.local_addr().unwrap();
387
388            for peer_addr in conn.paths_iter(local_addr) {
389                loop {
390                    let (write, send_info) = match conn.send_on_path(
391                        &mut out,
392                        Some(local_addr),
393                        Some(peer_addr),
394                    ) {
395                        Ok(v) => v,
396
397                        Err(quiche::Error::Done) => {
398                            break;
399                        },
400
401                        Err(e) => {
402                            log::error!(
403                                "{} -> {}: send failed: {:?}",
404                                local_addr,
405                                peer_addr,
406                                e
407                            );
408
409                            conn.close(false, 0x1, b"fail").ok();
410                            break;
411                        },
412                    };
413
414                    if let Err(e) = socket.send_to(&out[..write], send_info.to) {
415                        if e.kind() == std::io::ErrorKind::WouldBlock {
416                            log::debug!(
417                                "{} -> {}: send() would block",
418                                local_addr,
419                                send_info.to
420                            );
421                            break;
422                        }
423
424                        return Err(ClientError::Other(format!(
425                            "{} -> {}: send() failed: {:?}",
426                            local_addr, send_info.to, e
427                        )));
428                    }
429                }
430            }
431        }
432
433        if conn.is_closed() {
434            log::info!(
435                "connection closed, {:?} {:?}",
436                conn.stats(),
437                conn.path_stats().collect::<Vec<quiche::PathStats>>()
438            );
439
440            if !conn.is_established() {
441                log::info!(
442                    "connection timed out after {:?}",
443                    app_data_start.elapsed(),
444                );
445
446                return Err(ClientError::HandshakeFail);
447            }
448
449            break;
450        }
451    }
452
453    #[allow(irrefutable_let_patterns)]
454    let ClientVariant::Sync { stream_map, .. } = client_variant
455    else {
456        unimplemented!("async client supplied to sync client code");
457    };
458
459    Ok(ConnectionSummary {
460        stream_map,
461        stats: Some(conn.stats()),
462        path_stats: conn.path_stats().collect(),
463        conn_close_details: ConnectionCloseDetails::new(&conn),
464    })
465}

467fn check_duration_and_do_actions(
468    wait_duration: &mut Option<Duration>, wait_instant: &mut Option<Instant>,
469    action_iter: &mut Iter<Action>, conn: &mut quiche::Connection,
470    waiting_for: &mut WaitingFor, stream_parsers: &mut StreamParserMap,
471) {
472    match wait_duration.as_ref() {
473        None => {
474            if let Some(idle_wait) =
475                handle_actions(action_iter, conn, waiting_for, stream_parsers)
476            {
477                *wait_duration = Some(idle_wait);
478                *wait_instant = Some(Instant::now());
479
480                // TODO: the wait period could still be larger than the
481                // negotiated idle timeout.
482                // We could in theory check quiche's idle_timeout value if
483                // it was public.
484                log::info!(
485                    "waiting for {:?} before executing more actions",
486                    idle_wait
487                );
488            }
489        },
490
491        Some(period) => {
492            let now = Instant::now();
493            let then = wait_instant.unwrap();
494            log::debug!(
495                "checking if actions wait period elapsed {:?} > {:?}",
496                now.duration_since(then),
497                wait_duration
498            );
499            if now.duration_since(then) >= *period {
500                log::debug!("yup!");
501                *wait_duration = None;
502
503                if let Some(idle_wait) =
504                    handle_actions(action_iter, conn, waiting_for, stream_parsers)
505                {
506                    *wait_duration = Some(idle_wait);
507                }
508            }
509        },
510    }
511}

513/// Generate a new pair of Source Connection ID and reset token.
514pub fn generate_cid_and_reset_token() -> (quiche::ConnectionId<'static>, u128) {
515    let mut scid = [0; quiche::MAX_CONN_ID_LEN];
516    rand::RngCore::fill_bytes(&mut rand::thread_rng(), &mut scid);
517    let scid = scid.to_vec().into();
518    let mut reset_token = [0; 16];
519    rand::RngCore::fill_bytes(&mut rand::thread_rng(), &mut reset_token);
520    let reset_token = u128::from_be_bytes(reset_token);
521    (scid, reset_token)
522}

524/// Makes a buffered writer for a qlog.
525pub fn make_qlog_writer(
526    dir: &std::ffi::OsStr, role: &str, id: &str,
527) -> std::io::BufWriter<std::fs::File> {
528    let mut path = std::path::PathBuf::from(dir);
529    let filename = format!("{role}-{id}.sqlog");
530    path.push(filename);
531
532    match std::fs::File::create(&path) {
533        Ok(f) => std::io::BufWriter::new(f),
534
535        Err(e) => panic!(
536            "Error creating qlog file attempted path was {:?}: {}",
537            path, e
538        ),
539    }
540}

542fn handle_actions<'a, I>(
543    iter: &mut I, conn: &mut quiche::Connection, waiting_for: &mut WaitingFor,
544    stream_parsers: &mut StreamParserMap,
545) -> Option<Duration>
546where
547    I: Iterator<Item = &'a Action>,
548{
549    if !waiting_for.is_empty() {
550        log::debug!(
551            "won't fire an action due to waiting for responses: {:?}",
552            waiting_for
553        );
554        return None;
555    }
556
557    // Send actions
558    for action in iter {
559        match action {
560            Action::FlushPackets => return None,
561            Action::Wait { wait_type } => match wait_type {
562                WaitType::WaitDuration(period) => return Some(*period),
563                WaitType::StreamEvent(response) => {
564                    log::info!(
565                        "waiting for {:?} before executing more actions",
566                        response
567                    );
568                    waiting_for.add_wait(response);
569                    return None;
570                },
571            },
572            action => execute_action(action, conn, stream_parsers),
573        }
574    }
575
576    None
